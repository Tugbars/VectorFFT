# ==============================================================================
# VectorFFT - Radix-16 Tests
# ==============================================================================
# Compiler: Intel ICX
# All test names prefixed with "radix16_" for clarity in ctest output.
# ==============================================================================

cmake_minimum_required(VERSION 3.20)

enable_testing()

# ------------------------------------------------------------------------------
# Common compile flags
# ------------------------------------------------------------------------------

set(RADIX16_TEST_COMPILE_FLAGS
    -O2
    -Wall
    -Wextra
    -Wpedantic
    -Wno-unused-parameter
    -std=c17
)

# Per-ISA flags
set(RADIX16_FLAGS_SCALAR   -mno-sse)
set(RADIX16_FLAGS_AVX2     -march=haswell)
set(RADIX16_FLAGS_AVX512   -march=icelake-server)

# Common link libraries
set(RADIX16_TEST_LINK_LIBS
    vectorfft_radix16
    m
)

# ------------------------------------------------------------------------------
# Helper: add a radix16 test
# ------------------------------------------------------------------------------

function(add_radix16_test TEST_SUFFIX SOURCE_FILE ISA_FLAGS)
    set(TEST_TARGET  "test_radix16_${TEST_SUFFIX}")
    set(TEST_NAME    "radix16_${TEST_SUFFIX}")

    add_executable(${TEST_TARGET} ${SOURCE_FILE})

    target_include_directories(${TEST_TARGET} PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/..
        ${CMAKE_CURRENT_SOURCE_DIR}/../scalar
        ${CMAKE_CURRENT_SOURCE_DIR}/../sse2
        ${CMAKE_CURRENT_SOURCE_DIR}/../avx2
        ${CMAKE_CURRENT_SOURCE_DIR}/../avx512
    )

    target_compile_options(${TEST_TARGET} PRIVATE
        ${RADIX16_TEST_COMPILE_FLAGS}
        ${ISA_FLAGS}
    )

    target_link_libraries(${TEST_TARGET} PRIVATE ${RADIX16_TEST_LINK_LIBS})

    add_test(NAME ${TEST_NAME} COMMAND ${TEST_TARGET})
endfunction()

# ------------------------------------------------------------------------------
# Helper: add a radix16 benchmark (not registered as ctest by default)
# ------------------------------------------------------------------------------

function(add_radix16_bench BENCH_SUFFIX SOURCE_FILE ISA_FLAGS)
    set(BENCH_TARGET "bench_radix16_${BENCH_SUFFIX}")

    add_executable(${BENCH_TARGET} ${SOURCE_FILE})

    target_include_directories(${BENCH_TARGET} PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/..
        ${CMAKE_CURRENT_SOURCE_DIR}/../scalar
        ${CMAKE_CURRENT_SOURCE_DIR}/../sse2
        ${CMAKE_CURRENT_SOURCE_DIR}/../avx2
        ${CMAKE_CURRENT_SOURCE_DIR}/../avx512
    )

    target_compile_options(${BENCH_TARGET} PRIVATE
        ${RADIX16_TEST_COMPILE_FLAGS}
        -O3
        ${ISA_FLAGS}
    )

    target_link_libraries(${BENCH_TARGET} PRIVATE ${RADIX16_TEST_LINK_LIBS})
endfunction()

# ==============================================================================
# Tests
# ==============================================================================

# --- Scalar -------------------------------------------------------------------
add_radix16_test(scalar       test_radix16_scalar.c       "${RADIX16_FLAGS_SCALAR}")

# --- AVX2 --------------------------------------------------------------------
add_radix16_test(avx2         test_radix16_avx2.c         "${RADIX16_FLAGS_AVX2}")

# --- AVX-512 -----------------------------------------------------------------
add_radix16_test(avx512       test_radix16_avx512.c       "${RADIX16_FLAGS_AVX512}")

# --- Butterfly (cross-ISA / structural) --------------------------------------
add_radix16_test(butterfly    test_radix16_butterfly.c    "${RADIX16_FLAGS_AVX512}")

# ==============================================================================
# Benchmarks
# ==============================================================================

# --- Internal benchmark -------------------------------------------------------
add_radix16_bench(internal    bench_radix16.c             "${RADIX16_FLAGS_AVX512}")

# --- FFTW comparison ----------------------------------------------------------
find_library(FFTW3_LIB fftw3)
if(FFTW3_LIB)
    add_radix16_bench(vs_fftw bench_vs_fftw.c             "${RADIX16_FLAGS_AVX512}")
    target_link_libraries(bench_radix16_vs_fftw PRIVATE ${FFTW3_LIB})
else()
    message(STATUS "libfftw3 not found - skipping bench_vs_fftw")
endif()
